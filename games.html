let points = 0; let x = window.innerWidth / 2; let y = window.innerHeight / 2; let autoRate = 0; let costs = { turbine: 10, solar: 50 }; const player = document.getElementById('player'); const pointsDisplay = document.getElementById('points'); const rateDisplay = document.getElementById('auto-rate'); // Track which keys are currently held down const keys = {}; window.addEventListener('keydown', (e) => keys[e.key.toLowerCase()] = true); window.addEventListener('keyup', (e) => keys[e.key.toLowerCase()] = false); // SMOOTH MOVEMENT ENGINE function update() { const speed = 5; // Pixels per frame if (keys['w'] || keys['arrowup']) y -= speed; if (keys['s'] || keys['arrowdown']) y += speed; if (keys['a'] || keys['arrowleft']) x -= speed; if (keys['d'] || keys['arrowright']) x += speed; // Boundary checks (stay on screen) x = Math.max(0, Math.min(window.innerWidth - 50, x)); y = Math.max(80, Math.min(window.innerHeight - 50, y)); player.style.transform = `translate(${x}px, ${y}px)`; checkCollision(); requestAnimationFrame(update); } // Spawn Trash function spawnTrash() { const trash = document.createElement('div'); trash.className = 'trash'; trash.innerHTML = '🗑️'; trash.style.left = Math.random() * (window.innerWidth - 50) + 'px'; trash.style.top = Math.random() * (window.innerHeight - 150) + 100 + 'px'; document.getElementById('game-world').appendChild(trash); } // Collision Logic function checkCollision() { const trashes = document.querySelectorAll('.trash'); const playerRect = player.getBoundingClientRect(); trashes.forEach(t => { const trashRect = t.getBoundingClientRect(); if (!(playerRect.right < trashRect.left || playerRect.left > trashRect.right || playerRect.bottom < trashRect.top || playerRect.top > trashRect.bottom)) { t.remove(); points += 1; updateUI(); spawnTrash(); } }); } function buyUpgrade(type) { if (points >= costs[type]) { points -= costs[type]; const b = document.createElement('div'); b.className = 'building'; b.innerHTML = type === 'turbine' ? '🌬️
Turbine' : '☀️
Solar'; // Place building where player is standing b.style.left = x + 'px'; b.style.top = y + 'px'; document.getElementById('game-world').appendChild(b); autoRate += (type === 'turbine' ? 1 : 5); costs[type] = Math.round(costs[type] * 1.5); updateUI(); } else { alert("Collect more trash for Eco-Points!"); } } function updateUI() { pointsDisplay.innerText = points; rateDisplay.innerText = autoRate; document.getElementById('cost-turbine').innerText = costs.turbine; document.getElementById('cost-solar').innerText = costs.solar; } // Run auto-points every second setInterval(() => { points += autoRate; updateUI(); }, 1000); // Initialize for(let i=0; i<10; i++) spawnTrash(); update(); // Start the smooth movement loop
